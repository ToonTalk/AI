Raindrops Falling Moving Clouds let birds = []; function setup() { createCanvas(windowWidth, windowHeight); // Create birds at random positions for (let i = 0; i < 3; i++) { birds.push(new Bird(random(width), random(height / 3), random(20, 30), random(1, 2))); } } function draw() { background(135, 206, 250); // Sky blue background for (let bird of birds) { bird.update(); } } class Bird { constructor(x, y, size, speed) { this.x = x; this.y = y; this.size = size; this.speed = speed; this.wingAngle = 0; } update() { this.x += this.speed; this.wingAngle += 0.1; // Simulate wing flapping if (this.x > width) { this.x = -this.size; // Reset to left when bird goes off screen } this.display(); } display() { fill(0); noStroke(); // Draw bird body (ellipse) ellipse(this.x, this.y, this.size, this.size * 0.6); // Draw wings push(); translate(this.x, this.y); rotate(sin(this.wingAngle) * 0.2); // Flapping effect line(0, 0, this.size, -this.size / 1.5); rotate(PI); // Flip wing line(0, 0, this.size, -this.size / 1.5); pop(); } } Smaller Clouds
