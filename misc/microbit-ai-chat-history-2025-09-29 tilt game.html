from microbit import *
import random
--- Game Variables ---

player_x = 2  # Starting X position of the player (center)
player_y = 2  # Starting Y position of the player (center)
target_x = 0  # X position of the target
target_y = 0  # Y position of the target
score = 0     # Player's score
--- Function to generate a new target ---

Ensures the new target is not on the player's current position

def generate_new_target():
    global target_x, target_y # Declare that we are modifying the global variables
    while True:
        new_target_x = random.randint(0, 4) # Random X from 0 to 4
        new_target_y = random.randint(0, 4) # Random Y from 0 to 4
        # Make sure the new target is not the same as the player's current position
        if new_target_x != player_x or new_target_y != player_y:
            target_x = new_target_x
            target_y = new_target_y
            break # Exit the loop once a valid target is found
--- Game Initialization ---

generate_new_target() # Set the first target
display.scroll("Tilt to move!", 80) # Welcome message
sleep(500) # Short pause
--- Main Game Loop ---

while True:
    # 1. Read Accelerometer Input
    # Accelerometer values range roughly from -1024 to 1024 for X and Y tilt.
    # A value of 0 means flat.
    acc_x = accelerometer.get_x() # Read tilt along the X-axis (left/right)
    acc_y = accelerometer.get_y() # Read tilt along the Y-axis (up/down)
    # 2. Player Movement Logic
    # Adjust these threshold values (e.g., -200, 200) to change sensitivity.
    # Smaller values make it more sensitive, larger values less sensitive.
    # Move horizontally (left/right)
    if acc_x < -200: # Tilted significantly to the left
        player_x -= 1
    elif acc_x > 200: # Tilted significantly to the right
        player_x += 1
    # Move vertically (up/down)
    if acc_y < -200: # Tilted significantly upwards (towards the buttons)
        player_y -= 1
    elif acc_y > 200: # Tilted significantly downwards (away from the buttons)
        player_y += 1
    # 3. Boundary Checking
    # Ensure the player pixel stays within the 5x5 LED display (coordinates 0-4)
    player_x = max(0, min(4, player_x))
    player_y = max(0, min(4, player_y))
    # 4. Win Condition Check
    if player_x == target_x and player_y == target_y:
        score += 1 # Increase score
        # Display the score, converting the number to a string for concatenation
        display.scroll("Score: " + str(score), 80)
        generate_new_target() # Generate a new target for the next round
        sleep(200) # Short pause to acknowledge reaching the target
    # 5. Display Update
    display.clear() # Clear the entire screen
    display.set_pixel(player_x, player_y, 9) # Draw player (brightness 9, max)
    display.set_pixel(target_x, target_y, 4) # Draw target (brightness 4, medium)
    # 6. Game Speed Control
    # This sleep determines how often the game updates.
    # A smaller value makes the game more responsive but can feel "twitchy".
    # A larger value makes it slower and smoother.
    sleep(100) # Update every 100 milliseconds
